//BACKEND

1.npm init -y

2.npm i express bcryptjs cors dotenv jsonwebtoken mongoose multer socket.io

3.npm i nodemon --save-dev

4.package.json:

"scripts" : { "start": "node server.js", "dev": "nodemon server.js"}


5.Create your own .env file in backend folder


6.Run dev: npm run dev

2.npm install react-router-dom react-hot-toast framer-motion lucide-react moment

3.cd frontend ---- npm run dev
